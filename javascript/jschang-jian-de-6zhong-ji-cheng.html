    继承探究与实现

    # 继承探究与实现

    简单来说：继承是使子类具有父类的属性和方法。

    # 原型链继承

    原型链继承是比较常见的继承方式之一，其中涉及的构造函数、原型和实例，三者之间存在着一定的关系，即每一个构造函数都有一个原型对象，原型对象又包含一个指向构造函数的指针，而实例则包含一个原型对象的指针。

    	function Parent1() {
		this.name = 'parent1';
		this.play = [1,2,3];
	}
	function Child1() {
		this.type = 'child1';
	}
	Child1.prototype = new Parent1();
	console.log(new Child1());

    # 组合继承

    组合继承就是前两种继承方式的组合。

    	function Parent3 () {
		this.name = "parent3";
	}
	Parent3.prototype.getName = function() {
		return this.name;
	}
	function Child3 () {
		Parent3.call(this);
		this.name = 'child3'
	}
	Child3.prototype = new Parent3();
	//Child3.prototype会从原构造函数上继承一个constructor
	//所以手动挂上构造器，改回指向自己的构造函数
	Child3.prototype.constructor = Child3;
	let child = new Child3();
	console.log(child.name);
	console.log(child.getName());

    # 原型继承

     通过`Object.create()`实现，Object.create()会创建一个新的对象，此方法传入两个参数，一个参数指定新对象的原型对象，第二个参数为新对象添加（新属性）属性描述符（可选）；

       let parent4 = {
   name: "parent4",
   friends: ["p1", "p2", "p3"],
   getName: function() {
     return this.name;
   }
 };
   let parent5 = Object.create(parent4);
   parent5.name = "Tom";
   parent5.friends.push("p5");
   
   let parent6 = Object.create(parent4);
   parent5.name = "Jerry";
   (parent5.friends).push("p6");
   console.log("parent5.name:",parent5.name)
   console.log("parent6.name:",parent6.name)
   console.log(parent5.friends)
   console.log(parent6.friends)
   console.log(parent5.name === parent5.getName())
   console.log(parent6.name === parent6.getName())

    # 寄生式继承

    使用原型式继承可以获得一份目标对象的浅拷贝，然后利用这个浅拷贝的能力再进行增强，添加一些方法，这样的继承方式就叫作寄生式继承。

    let parent5 = {
	name: "parent5",
	type: [1,2,3],
	getName: function() {
		return this.name;
	}
}
function clone(origin) {
	let clone = Object.create(origin);
	clone.getType = function() {
		return this.type;
	}
	return clone;
}
let person5 = clone(parent5);
person5.type.push(4);
console.log(person5.getName())
console.log(person5.getType())

    # 寄生组合继承

    结合寄生继承和组合继承，在前几种继承方式的优缺点的基础上进行改造，得出寄生组合继承：

    function clone(parent, child) {
	//这里替换了 new parent()
	child.prototype = Object.create(parent.prototype);
	child.prototype.constructor = child;
}
function Parent6(){
	this.name = "parent6";
	this.type = [1,2,3];
}
Parent6.prototype.getName = function() {
	return this.name;
}
function Child6() {
	//定义上下文this
	Parent6.call(this);
	this.friend = "child5";
}
clone(Parent6, Child6);
Child6.prototype.getFriends = function() {
	return this.friend;
}
let person6 = new Child6();
console.log(person6);
console.log(person6.getName());
console.log(person6.getFriends());
